def request_panel_date(root):
    date_map = {}
    for file, base_dir, source in get_loading_info(root):
        with closing(load_panels(file, base_dir)) as panels:
            for panel in panels:
                date_map[panel.date] = panel.get_rating(default=None)
    return tools.DateRequester(calendar.MONDAY).request(date_map.items())


def load_single_panel(root, date):
    for panel in load_merged_panels(root):
        if panel.date == date:
            return panel
    raise ValueError(f'cannot find panel on {date}')


def load_merged_panels(root):
    panel_map = collections.defaultdict(list)
    for file, base_dir, source in get_loading_info(root):
        with closing(load_panels(file, base_dir)) as panels:
            for panel in panels:
                panel_map[panel.date].append((panel, source))
    for date, values in panel_map.items():
        tools.check_panel_attributes(values)
        yield tools.merge_panels(panel for panel, _ in values)


def load_panels(file, base_dir):
    loader = mystuff.JSONLoader(base_dir=base_dir, error_on_warning=True)
    with open_for_reading(file) as fp:
        yield from loader.load(fp)


def get_loading_info(root):
    root = Path(root)
    data_dir = root / DATA_DIRNAME
    base_dir = root / BASE_DIRNAME
    with open_for_reading(data_dir / 'sources.txt') as fp:
        sources = fp.read().splitlines()
    for source in sources:
        yield data_dir / source, base_dir, source


def get_description(file, base_dir):
    loader = mystuff.JSONLoader(base_dir=base_dir, error_on_warning=True)
    with open_for_reading(file) as fp:
        data = loader.load_json(fp)
    _, attrs = loader.split_data(data)
    try:
        return attrs['desc']
    except KeyError:
        return None

def load_migration_script(path):
    # XXX: wtf???
    for modname in 'config', 'mystuff', 'tools':
        if modname in sys.modules:
            del sys.modules[modname]
    if os.path.isdir(path):
        config_path = os.path.join(path, 'scripts', 'config.py')
        if not os.path.exists(config_path):
            raise ValueError(f'project directory {path!r} does not '
                             f'have scripts/config.py')
        # name shadowing >_>
        config_mod = tools.load_module_from_file('config', config_path)
        return config_mod.load_merged_panels(path)
    else:
        migrate_mod = tools.load_module_from_file('migrate', path)
        return migrate_mod.get_panels()


def merge_panels(root, new_panels):
    panel_map = collections.defaultdict(list)
    for panel, source in itertools.chain(
            zip(new_panels,
                itertools.repeat('external source')),
            zip(load_merged_panels(root),
                itertools.repeat('current project'))):
        panel_map[panel.date].append((panel, source))
    merged = []
    for date, values in panel_map.items():
        tools.check_panel_attributes(values)
        yield tools.merge_panels(panel for panel, _ in values)


def dump_panels(root, panels):
    root = Path(root)
    data_dir = root / DATA_DIRNAME
    base_dir = root / BASE_DIRNAME
    data_dir.mkdir()
    with open_for_writing(data_dir / 'sources.txt') as fp:
        fp.write('backups/1.json\n')
    (data_dir / 'backups').mkdir()
    dumper = mystuff.MyJSONDumper()
    # sort all panels and entries because we don't want JSONLoader's
    # check options emitting warnings
    panels = sorted(panels, key=lambda p: p.date)
    for panel in panels:
        panel.sort_entries(key=lambda e: e.time.astimezone(timezone.utc))
    dumper.configure(base_dir=base_dir, error_on_warning=True)
    with open_for_writing(data_dir / 'backups' / '1.json') as fp:
        dumper.dump(panels, fp)


def clean_up_data(root):
    root = Path(root)
    data_dir = root / DATA_DIRNAME
    base_dir = root / BASE_DIRNAME
    if data_dir.exists():
        shutil.rmtree(data_dir)
    if base_dir.exists():
        shutil.rmtree(base_dir)


def open_for_reading(file):
    return open(file, encoding=FILE_ENCODING)


def open_for_writing(file):
    return open(file, 'x', encoding=FILE_ENCODING)

def list_backups(root):
    try:
        return sorted((entry.name for entry in
                       os.scandir(os.path.join(root, BACKUP_NAME))
                       if entry.is_dir()))
    except FileNotFoundError:
        return []


def create_backup(root, name):
    backup_dir = os.path.join(root, BACKUP_NAME, name)
    os.makedirs(backup_dir)
    with os.scandir(root) as it:
        for entry in it:
            if entry.name == BACKUP_NAME or entry.name in IGNORE_NAMES:
                continue
            src = entry.path
            dst = os.path.join(backup_dir, entry.name)
            if entry.is_dir():
                shutil.copytree(src, dst)
            else:
                shutil.copy2(src, dst)
    return os.path.join(BACKUP_NAME, name)


def show_backup_diff(root, name):
    backup_dir = os.path.join(root, BACKUP_NAME, name)
    left, right = '.', os.path.join(BACKUP_NAME, name)
    dcmp = filecmp.dircmp(os.path.join(root, left),
                          os.path.join(root, right))
    return _diffdirs_impl(root, dcmp, left, right)


def restore_backup(root, name):
    backup_dir = os.path.join(root, BACKUP_NAME, name)
    if not os.path.isdir(backup_dir):
        raise ValueError(f'cannot find backup {name!r}')
    with os.scandir(root) as it:
        for entry in it:
            if entry.name == BACKUP_NAME or entry.name in IGNORE_NAMES:
                continue
            if entry.is_dir():
                shutil.rmtree(entry)
            else:
                os.unlink(entry)
    with os.scandir(backup_dir) as it:
        for entry in it:
            src = entry.path
            dst = os.path.join(root, entry.name)
            if entry.is_dir():
                shutil.copytree(src, dst)
            else:
                shutil.copy2(src, dst)


def remove_backup(root, name):
    backup_dir = os.path.join(root, BACKUP_NAME, name)
    shutil.rmtree(backup_dir)

def select_backup_name(root):
    backups = list_backups(root)
    if not backups:
        raise ValueError("You haven't created any backups!")
    print('Choose from one of the following backups:')
    for i, backup in enumerate(backups, start=1):
        print(f'  ({i}) {backup}')
    while True:
        s = input('Number: ')
        try:
            selected = backups[int(s) - 1]

            print(f"{s!r} isn't a valid backup number", file=sys.stderr)
        else:
            print(f'Selected {selected!r}')
            return selected


def backup_exists(root, name):
    return os.path.exists(os.path.join(root, BACKUP_NAME, name))


def generate_backup_name(root):
    base = datetime.now().astimezone().strftime('%Y-%m-%dT%H-%M-%S%z')
    name = base
    i = 1
    while backup_exists(root, name):
        name = f'{base}_{i}'
        i += 1
    return name


#
# borrowed from these:
#  *  https://docs.python.org/3/library/difflib.html#a-command-line-interface-to-difflib
#  *  https://docs.python.org/3/library/filecmp.html#filecmp.dircmp
#
# implementation, adapted to ignore BACKUP_NAME and IGNORE_NAMES
#
def _diffdirs_impl(root, dcmp, left, right):
    exit_status = 0
    for name in dcmp.diff_files:
        left_file = os.path.join(dcmp.left, name)
        right_file = os.path.join(dcmp.right, name)
        # picking a random side to get the relative path
        path = os.path.relpath(left_file, left)
        try:
            with open_for_reading(left_file) as fp:
                left_lines = fp.readlines()
            with open_for_reading(right_file) as fp:
                right_lines = fp.readlines()
        except UnicodeDecodeError:
            print(f'Binary files {path} in {left} and {right} differ')
        else:
            left_mtime = _file_mtime(left_file)
            right_mtime = _file_mtime(right_file)
            left_relpath = _relpath(left_file, start=root)
            right_relpath = _relpath(right_file, start=root)
            diff = difflib.unified_diff(left_lines, right_lines,
                                        left_relpath, right_relpath,
                                        left_mtime, right_mtime)
            print(f'Diff of {path} below:')
            sys.stdout.writelines(diff)
        exit_status = 1
    for left_name in dcmp.left_only:
        left_path = os.path.relpath(
            os.path.join(dcmp.left, left_name), left)
        if left_path == BACKUP_NAME or left_path in IGNORE_NAMES:
            continue
        print(f'Only in {left}: {left_path}')
        exit_status = 1
    for right_name in dcmp.right_only:
        right_path = os.path.relpath(
            os.path.join(dcmp.right, right_name), right)
        print(f'Only in {right}: {right_path}')
        exit_status = 1
    for sub_dcmp in dcmp.subdirs.values():
        exit_status |= _diffdirs_impl(root, sub_dcmp, left, right)
    return exit_status


def _file_mtime(path):
    t = datetime.fromtimestamp(os.stat(path).st_mtime, timezone.utc)
    return t.astimezone().isoformat()


# alternative version of os.path.relpath but doesn't normalize
# the path (very cheesy i know)
def _relpath(path, start):
    if path.startswith(start):
        return path[len(start):].lstrip(os.sep)
    return os.path.relpath(path, start)
